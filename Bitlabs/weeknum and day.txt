start
step1:read any weeknumber wn
step2:if(wn==1)
      { 
       display "Monday"
      }
      else if(wn==2)
      {
       display "Tuesday"
      }
      else if(wn==3)
      {
       display "wednesday"
      }
      else if(wn==4)
      {
       display "Thursday"
      }
      else if(wn==5)
      {
       display "Friday"
      }
      else if(wn==6)
      {
       display "saturday"
      }
      else if(wn==7)
      {
       display "Sunday"
      }
      else
      {
       display "invalid weeknumber"
      }
stop
